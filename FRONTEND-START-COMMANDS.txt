🚀 2. OTOMATİK BAŞLATMA (ÖNERİLEN)
═══════════════════════════════════════════════════════════════

cd /root/neuroviabot/bot/neuroviabot-discord
chmod +x START-FRONTEND-SAFE.sh
./START-FRONTEND-SAFE.sh


📝 3. MANUEL BAŞLATMA (ADIM ADIM)
═══════════════════════════════════════════════════════════════

# Adım 1: Klasöre git
cd /root/neuroviabot/bot/neuroviabot-frontend

# Adım 2: PM2 durumunu kontrol et
pm2 list

# Adım 3a: Frontend ÇALIŞIYORsa (restart)
pm2 restart neuroviabot-frontend

# Adım 3b: Frontend ÇALIŞMIYORSA (start)
pm2 start "node_modules/.bin/next start -p 3001" --name "neuroviabot-frontend"

# Adım 4: Save
pm2 save

# Adım 5: Logları kontrol et
pm2 logs neuroviabot-frontend --lines 20


🔍 4. SORUN GİDERME
═══════════════════════════════════════════════════════════════

# Problem: "next: not found"
# Çözüm: node_modules kurulu mu kontrol et
ls -la node_modules/.bin/next

# Eğer next yoksa:
cd /root/neuroviabot/bot/neuroviabot-frontend
npm install --legacy-peer-deps


# Problem: ".next klasörü yok"
# Çözüm: Build yap
cd /root/neuroviabot/bot/neuroviabot-frontend
npm run build


# Problem: "Port 3001 already in use"
# Çözüm: Eski process'i durdur
pm2 delete neuroviabot-frontend
lsof -ti:3001 | xargs kill -9
pm2 start "node_modules/.bin/next start -p 3001" --name "neuroviabot-frontend"


# Problem: Frontend dosyaları SİLİNMİŞ
# Çözüm: Git'ten clone et
cd /root/neuroviabot/bot
git clone https://github.com/YOUR_USERNAME/neuroviabot-discord.git temp
cp -r temp/neuroviabot-frontend/* neuroviabot-frontend/
rm -rf temp
cd neuroviabot-frontend
npm install --legacy-peer-deps
npm run build


🔧 5. FULL REBUILD (Son Çare - SİLER VE YENİDEN YAPAR)
═══════════════════════════════════════════════════════════════

cd /root/neuroviabot/bot/neuroviabot-frontend

# 1. PM2'den kaldır
pm2 delete neuroviabot-frontend

# 2. Build cache'i temizle
rm -rf .next

# 3. Build yap
npm run build

# 4. Start
pm2 start "node_modules/.bin/next start -p 3001" --name "neuroviabot-frontend"

# 5. Save
pm2 save
